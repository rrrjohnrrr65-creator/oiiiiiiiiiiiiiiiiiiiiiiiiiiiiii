    <script>
        function openBook() {
            const book = document.querySelector('.book-container');
            const music = document.getElementById('bg-music');
            const stickers = document.querySelectorAll('.sticker');
            
            // Check if already open to prevent double clicks
            if (!book.classList.contains('open')) {
                book.classList.add('open');
                
                // Play Music (Set volume softer)
                music.volume = 0.5; 
                music.play().then(() => {
                    console.log("Music playing!");
                }).catch(e => {
                    console.log("Tap required for audio");
                });

                // Show outside stickers with a little delay
                setTimeout(() => {
                    stickers.forEach(s => s.style.display = 'block');
                }, 400);

                // Start Heart Rain Effect
                for(let i=0; i<30; i++) {
                    setTimeout(createHeart, i * 200);
                }
            }
        }

        function createHeart() {
            const heart = document.createElement('div');
            heart.classList.add('emoji-rain');
            // Random cute emojis
            const emojis = ['üíó', '‚ú®', 'üê±', 'üå∏', 'üíñ'];
            heart.innerText = emojis[Math.floor(Math.random() * emojis.length)];
            
            heart.style.left = Math.random() * 100 + 'vw';
            heart.style.animationDuration = Math.random() * 2 + 3 + 's';
            
            document.getElementById('hearts-container').appendChild(heart);
            
            setTimeout(() => {
                heart.remove();
            }, 5000);
        }
